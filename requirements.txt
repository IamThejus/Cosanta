# Cosanta — pip dependencies (Android / Termux, aarch64)
#
# These are the ONLY packages that should be installed with pip. They are all
# pure Python and compile nothing. Everything with native code (numpy, scipy,
# onnxruntime, cffi, portaudio, whisper.cpp, Android TTS) comes from `pkg` —
# see DEPENDENCIES.md and run ./setup.sh, which installs those first.
#
# NOTE: openwakeword is installed SEPARATELY with `--no-deps` (setup.sh does
# this) so pip does not drag in tflite-runtime / scipy / onnxruntime builds.
# It is intentionally NOT listed here.

sounddevice>=0.4.6   # Mic capture over PortAudio. Pure Python; uses the pkg
                     #   'python-cffi' + 'portaudio'. Only lib that streams raw
                     #   PCM frames, which OpenWakeWord needs (termux-mic-record
                     #   writes files and cannot stream).
requests>=2.31       # Groq REST client + model downloads. Pure Python. Chosen
                     #   over the groq SDK, which needs pydantic-core (Rust).
tqdm>=4.65           # Progress bars for OpenWakeWord's model downloads. Pure
                     #   Python; an OpenWakeWord runtime import.
python-dotenv>=1.0   # Optional .env loader (a minimal fallback parser is built
                     #   in). Pure Python.
